contact={}
while True:
    print("Contact Management\n\n1-Add a contact\n2-View a contact\n3-Edit a contact\n4-Exit")
    choose=int(input("\nPlease choose a number for 1-4: "))
    if choose==1:
        user_id=input("Enter the contect ID: ")
        user_name=input("Please type a name: ")
        phone_number=input("Please type a phone number: ")
        contact["ID"]=user_id
        contact["name"]=user_name
        contact["phone"]=phone_number
        print(f"\n{user_name} was added successfully...")
    elif choose==2:
        print(contact)
    elif choose==3:
        id_to_edit=input("Please enter and ID to edit: ")
        if id_to_edit in contact["ID"]:
            new_name=input("Please type a name: ")
            new_phone_number=input("Please type a phone number: ")
            contact["name"]=new_name
            contact["phone"]=new_phone_number
            print(f"\n{new_name} was added successfully...")
        else:
            print(f"Sorry, {id_to_edit} was not found....")    
    elif choose==4:
        break
    else:
        print("invalid selecotion")
